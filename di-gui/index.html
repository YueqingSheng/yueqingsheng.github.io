class Solution:
    def maxDepth(self, root: TreeNode) -> int:
        if not root:
            return 0
        ldepth = self.maxDepth(root.left) + 1 
        rdepth = self.maxDepth(root.right) + 1
        return max(ldepth, rdepth)
